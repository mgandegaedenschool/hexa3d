<?php

namespace App\Models;

use App\Models\User;
use Illuminate\Support\Str;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Factories\HasFactory;

class ResponseA extends Model
{
    use HasFactory;
    protected $guarded = [];
    // protected $guarded = [];
    // protected $fillable = ['member_id'];

    public static function boot()
    {
        parent::boot();
        self::creating(function ($responseA) {
            // $post->user()->associate(auth()->user()->id);
            // $post->category()->associate(request()->category);
            $responseA->user()->associate(auth()->user()->id);
        });
        self::updating(function ($responseA) {
            $responseA->user()->associate(request()->user);
        });
    }
    public function user()
    {
        return $this->belongsTo(User::class);
    }
    public function getTitleAttribute($attribute)
    {
        return Str::title($attribute);
    }
}
